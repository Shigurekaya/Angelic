// [XXX]motion.tjsをロードする前に存在しなければならないのでここに配置
function CustomSystemExceptionHandlerHook(e, message) {
	var message = "";
	try { message = (e instanceof "Exception") ? ((string)e.message).toLowerCase() : ""; } catch {} // [MEMO] eがオブジェクトでない場合がある
	if (message.indexOf("direct3dcreate9") >= 0 || message.indexOf("motionplayer") >= 0) {
		var enKag = typeof global.kag == "Object";
		var text = (("DirectXのランタイムが古い／もしくはグラフィックドライバが古い可能性があります。\n"
					"ゲーム付属の「お読みください」のテキストにに従い、ランタイムやドライバを更新してみてください。\n"
					"更新しても改善されない場合は、ここでDirect3Dを使用しない設定（ただし表示負荷が高い）に変更することができます。"
					"（この設定は「ヘルプ／環境依存の不具合回避」で後から変更することができます）\n")
					+(@"【サポート用エラーメッセージ：${e.message}】\n")+
					("\n設定を変更" + (enKag?"してゲームを再起動しますか？":"しますか？")));
		if (System.confirm(text)) with (Storages) {
			// [XXX]
			try { Plugins.link("PackinOne.dll") if (typeof global.Array.save2 == "undefined"); } catch{}
			var path = Storages.getFullPath(System.dataPath + "/" + .chopStorageExt(.extractStorageName(System.exeName)) + ".cfu");
			var cfu;
			if (Storages.isExistentStorage(path)) {
				cfu = [];
				try { cfu.load(path); } catch { cfu = void; }
			}
			if (cfu === void) cfu = [
				'; ============================================================================',
				'; *DO NOT EDIT* this file unless you are understanding what you are doing.',
				'; FYI:',
				';  Each line consists of NAME="VALUE" pair, VALUE is a series of',
				';  \\xNN, where NN is hexadecimal representation of UNICODE codepoint.',
				';  For example, opt="\\x61\\x62\\x63\\x3042\\x3044\\x3046" means that the',
				';  value of options "opt" is alphabets a, b, and c followed by Japanese',
				';  Hiraganas A, I, and U.',
				';  DO NOT PUT non-escaped value like opt="abc". This doesn\'t work and should',
				';  be like opt="\\x61\\x62\\x63".',
				'; ============================================================================' ];

			var key = "nod3dm=", value = '"\\x79\\x65\\x73"', found = false; // "yes"
			for (var i = 0; i < cfu.count; i++) {
				if (((string)cfu[i]).substr(key.length) == key) {
					cfu[i] = !found ? (key + value) : void;
					found = true;
				}
			}
			if (found) cfu.remove(void, true);
			else cfu.add(key+value);

			try { cfu.save2(path); }
			catch { System.inform(@"設定を保存できませんでした: 書き込み失敗\n${Storages.getLocalName(path)}"); }

			if (enKag) systemReboot();
		}
		return true;
	}
}
